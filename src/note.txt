DECLARE @trnsDtS	VARCHAR(8) = '20191007'
DECLARE @startTime  VARCHAR(20) = '2000'
DECLARE @intervalTime  VARCHAR(20) = '5'
DECLARE @convertTime  VARCHAR(20) = substring(@trnsDtS+@startTime,1,8) + ' ' 
+ substring(@trnsDtS+@startTime,9,2) + ':' 
+ substring(@trnsDtS+@startTime,11,2)

select substring(@trnsDtS+@startTime,9,2) + substring(@trnsDtS+@startTime,11,2)
SELECT REPLACE(CONVERT(CHAR(5), DATEADD(MI,CONVERT(int, @intervalTime),@convertTime), 8),':','')
SELECT REPLACE(CONVERT(CHAR(5), DATEADD(MI,CONVERT(int, 2*@intervalTime),@convertTime), 8),':','')
